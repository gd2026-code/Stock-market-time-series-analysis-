from statsmodels.tsa.stattools import adfuller
def fullertest(x):
    result = adfuller(x.dropna())
    print("ADF Statistic:", result[0])
    print("p-value:", result[1])
    print("Critical Values:")
    for key, value in result[4].items():
        print(f"  {key}: {value}")
